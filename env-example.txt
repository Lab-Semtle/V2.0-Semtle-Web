# Supabase Configuration
# 1. Supabase 대시보드 (https://supabase.com/dashboard)에서 프로젝트 선택
# 2. Settings > API 메뉴로 이동
# 3. Project URL을 복사하여 아래에 붙여넣기
NEXT_PUBLIC_SUPABASE_URL=your_supabase_project_url_here

# 4. Project API keys 섹션에서 anon public 키를 복사하여 아래에 붙여넣기
NEXT_PUBLIC_SUPABASE_ANON_KEY=your_supabase_anon_key_here

# 5. Project API keys 섹션에서 service_role 키를 복사하여 아래에 붙여넣기
# 주의: 이 키는 서버에서만 사용되며 클라이언트에 노출되면 안됩니다
SUPABASE_SERVICE_ROLE_KEY=your_supabase_service_role_key_here

# Site URL Configuration
# 프로덕션 환경에서는 실제 도메인을 입력하세요
# 예: https://yourdomain.com
NEXT_PUBLIC_SITE_URL=http://localhost:3000
